def delete_item(self, item: Any) -> bool:
    if self.is_empty():
        return False  # tree is empty
    elif self._subtrees == []:
        if self._root != item:
            return False
        else:
            self._root = None
            return True
    else:
        if self._root == item:
            self._delete_root()  # promote a child
            return True
        else:
            for subtree in self._subtrees:
                subtree.delete_item(item)
