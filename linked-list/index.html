class Node:
    def __init__(self, data):
        self.data = data
        self.next = None

class LinkedList:
    def __init__(self):
        self.first = None

    def append(self, data):
        new_node = Node(data)
        if not self.first:
            self.first = new_node
            return
        last = self.first
        while last.next:
            last = last.next
        last.next = new_node

# Example usage
ll1 = LinkedList()
ll1.append(10)
ll1.append(20)
